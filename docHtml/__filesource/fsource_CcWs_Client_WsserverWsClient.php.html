  279.         // fetch message length

  281.

  283.         $bufferSize = 4096;

  293.  
  294.         // loop around all frames to send
  295.         for($i = 0; $i < $frameCount; $i++)

  334.                 $left -= $sent;

  336.                     $buffer = substr($buffer, $sent);
  337.             }

  339.         }

  341.         return true;
  342.     }

  373.     {

  376.         $sep = strpos($buffer, "\r\n\r\n");

  381.         $headersCount = sizeof($headers); // includes request line
  382.         if($headersCount < 1)
  383.             return false;
  384.  
  385.         // fetch request and check it has at least 3 parts (space tokens)

  389.         if($requestPartsSize < 3)
  390.             return false;
  391.  
  392.         // check request method is GET
  393.         if(strtoupper($requestParts[0]) != 'GET')
  394.             return false;
  395.  
  396.         // check request HTTP version is at least 1.1
  397.         $exp = explode('?', isset($requestParts[1]) ? $requestParts[1] : '');

  399.         unset($exp[0]);

  402.  
  403.  
  404.         $httpPart = &$requestParts[$requestPartsSize - 1];
  405.         $httpParts = explode('/', $httpPart);
  406.         if(!isset($httpParts[1]) || (float) $httpParts[1] < 1.1)
  407.             return false;
  408.  
  409.         // store headers into a keyed array: array[headerKey] = headerValue
  410.         $headersKeyed = array();
  411.         for($i = 1; $i < $headersCount; $i++)
  412.         {
  413.             $parts = explode(':', $headers[$i]);
  414.             if(!isset($parts[1]))
  415.                 return false;
  416.  
  417.             $headersKeyed[trim($parts[0])] = trim($parts[1]);
  418.         }
  419.         $this->HeadersRequest = $headersKeyed;
  420.         // check Host header was received
  421.         if(!isset($headersKeyed['Host']))
  422.             return false;
  423.

  430.  
  431.         // check Sec-WebSocket-Version header was received and value is 7

  433.             return false; // should really be != 7, but Firefox 7 beta users send 8
  434.  
  435.  
  436.  
  437.  
  438.  
  439.  
  440.  
  441.             
  442. // work out hash to use in Sec-WebSocket-Accept reply header

  447.             'HTTP/1.1 101 Switching Protocols',
  448.             'Upgrade: websocket',
  449.             'Connection: Upgrade',
  450.             'Sec-WebSocket-Accept: ' . $hash

  452.         $this->HeaderResponse = $headers;
  453.  
  454.         RETURN true;
  455.     }

  551.

  553.

  619.                     $this->MessageBufferLength = 0;

  650.      * @return boolean 
  651.      */
  652.     public function BuildFrame(&$buffer, $bufferLength)
  653.     {
  654.         // increase number of bytes read for the frame, and join buffer onto end of the frame buffer
  655.         $this->FrameBytesRead += $bufferLength;

  684.                     $this->FrameBytesRead = 0;

  689.                 if($nextFrameBytesLength <= 0 || !$result)

  734.

  742.                         return false;

  784.                     }

  786.

  788.             }
  789.         }
