  444.         // If the item has an alias declaration we remove it and set it aside.

  462.         }
  463.  
  464.         // Break the string apart if it contains periods, then insert the table prefix
  465.         // in the correct location, assuming the period doesn't indicate that we're dealing
  466.         // with an alias. While we're at it, we will escape the components

  472.             // one of the aliases previously identified?  If so,
  473.             // we have nothing more to do other than escape the item

  475.             {

  477.                 {
