  132.         $nextPingCheck = time() + 1;

  263.     private function wsCheckIdleClients()
  264.     {
  265.         $time = time();
  266.         /* @var $client WsClient */
  267.         foreach($this->wsClients as $clientID => &$client)
  268.         {
  269.             if($client->ReadyState != self::WS_READY_STATE_CLOSED)
  270.             {
  271.                 // client ready state is not closed
  272.                 if($client->PingSentTime !== false)
  273.                 {
  274.                     // ping request has already been sent to client, pending a pong reply
  275.                     if($time >= $client->PingSentTime + self::WS_TIMEOUT_PONG)
  276.                     {
  277.                         // client didn't respond to the server's ping request in self::WS_TIMEOUT_PONG seconds
  278.  
  279.                         $this->wsClients[$clientID]->SendClose(self::WS_STATUS_TIMEOUT);
  280.                         $this->wsClients[$clientID]->Remove();
  281.                     }
  282.                 } elseif($time >= $client->LastRecvTime + self::WS_TIMEOUT_RECV)
  283.                 {
  284.                     // last data was received >= self::WS_TIMEOUT_RECV seconds ago
  285.                     if($client->ReadyState != self::WS_READY_STATE_CONNECTING)
  286.                     {
  287.                         // client ready state is open or closing
  288.                         $this->wsClients[$clientID]->SetPingSentTime(time());
  289.                         $this->wsClients[$clientID]->SendMensaje(self::WS_OPCODE_PING, '');
  290.                     } else
  291.                     {
  292.                         // client ready state is connecting
  293.                         $this->wsClients[$clientID]->Remove();
  294.                     }
  295.                 }
  296.                 if($client instanceof WsClient)
  297.                     $this->CallEvent('OnLoopClient', [&$client]);

  299.         }
