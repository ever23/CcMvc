  152.     public function xss_clean($str, $is_image = FALSE)
  153.     {
  154.         // Is the string an array?
  155.         //Remove Invisible Characters
  156.         $str = $this->remove_invisible_characters($str);
  157.  
  158.         /*
  159.          * URL Decode
  160.          *
  161.          * Just in case stuff like this is submitted:
  162.          *
  163.          * <a href="http://%77%77%77%2E%67%6F%6F%67%6C%65%2E%63%6F%6D">Google</a>
  164.          *
  165.          * Note: Use rawurldecode() so it does not remove plus signs
  166.          */
  167.         do
  168.         {
  169.             $str = rawurldecode($str);
  170.         } while (preg_match('/%[0-9a-f]{2,}/i', $str));
  171.  
  172.         /*
  173.          * Convert character entities to ASCII
  174.          *
  175.          * This permits our tests below to work reliably.
  176.          * We only convert entities that are within tags since
  177.          * these are the ones that will pose security problems.
  178.          */
  179.         $str = preg_replace_callback("/[^a-z0-9>]+[a-z0-9]+=([\'\"]).*?\\1/si", array($this, '_convert_attribute'), $str);
  180.         $str = preg_replace_callback('/<\w+.*/si', array($this, '_decode_entity'), $str);
  181.  
  182.         // Remove Invisible Characters Again!
  183.         $str = $this->remove_invisible_characters($str);
  184.  
  185.         /*
  186.          * Convert all tabs to spaces
  187.          *
  188.          * This prevents strings like this: ja    vascript
  189.          * NOTE: we deal with spaces between characters later.
  190.          * NOTE: preg_replace was found to be amazingly slow here on
  191.          * large blocks of data, so we use str_replace.
  192.          */
  193.         $str = str_replace("\t", ' ', $str);
  194.  
  195.         // Capture converted string for later comparison
  196.         $converted_string = $str;
  197.  
  198.         // Remove Strings that are never allowed
  199.         $str = $this->_do_never_allowed($str);
  200.  
  201.         /*
  202.          * Makes PHP tags safe
  203.          *
  204.          * Note: XML tags are inadvertently replaced too:
  205.          *
  206.          * <?xml
  207.          *
  208.          * But it doesn't seem to pose a problem.
  209.          */
  210.         if ($is_image === TRUE)
  211.         {
  212.             // Images have a tendency to have the PHP short opening and
  213.             // closing tags every so often so we skip those and only
  214.             // do the long opening tags.
  215.             $str = preg_replace('/<\?(php)/i', '&lt;?\\1', $str);
  216.         } else
  217.         {
  218.             $str = str_replace(array('<?', '?' . '>'), array('&lt;?', '?&gt;'), $str);
  219.         }
  220.  
  221.         /*
  222.          * Compact any exploded words
  223.          *
  224.          * This corrects words like:  j a v a s c r i p t
  225.          * These words are compacted back to their correct state.
  226.          */
  227.         $words = array(
  228.             'javascript', 'expression', 'vbscript', 'jscript', 'wscript',
  229.             'vbs', 'script', 'base64', 'applet', 'alert', 'document',
  230.             'write', 'cookie', 'window', 'confirm', 'prompt'
  231.         );
  232.  
  233.         foreach ($words as $word)
  234.         {
  235.             $word = implode('\s*', str_split($word)) . '\s*';
  236.  
  237.             // We only want to do this when it is followed by a non-word character
  238.             // That way valid stuff like "dealer to" does not become "dealerto"
  239.             $str = preg_replace_callback('#(' . substr($word, 0, -3) . ')(\W)#is', array($this, '_compact_exploded_words'), $str);
  240.         }
  241.  
  242.         /*
  243.          * Remove disallowed Javascript in links or img tags
  244.          * We used to do some version comparisons and use of stripos(),
  245.          * but it is dog slow compared to these simplified non-capturing
  246.          * preg_match(), especially if the pattern exists in the string
  247.          *
  248.          * Note: It was reported that not only space characters, but all in
  249.          * the following pattern can be parsed as separators between a tag name
  250.          * and its attributes: [\d\s"\'`;,\/\=\(\x00\x0B\x09\x0C]
  251.          * ... however, remove_invisible_characters() above already strips the
  252.          * hex-encoded ones, so we'll skip them below.
  253.          */
  254.         do
  255.         {
  256.             $original = $str;
  257.  
  258.             if (preg_match('/<a/i', $str))
  259.             {
  260.                 $str = preg_replace_callback('#<a[^a-z0-9>]+([^>]*?)(?:>|$)#si', array($this, '_js_link_removal'), $str);
  261.             }
  262.  
  263.             if (preg_match('/<img/i', $str))
  264.             {
  265.                 $str = preg_replace_callback('#<img[^a-z0-9]+([^>]*?)(?:\s?/?>|$)#si', array($this, '_js_img_removal'), $str);
  266.             }
  267.  
  268.             if (preg_match('/script|xss/i', $str))
  269.             {
  270.                 $str = preg_replace('#</*(?:script|xss).*?>#si', '[removed]', $str);
  271.             }
  272.         } while ($original !== $str);
  273.  
  274.         unset($original);
  275.  
  276.         // Remove evil attributes such as style, onclick and xmlns
  277.         $str = $this->_remove_evil_attributes($str, $is_image);
  278.  
  279.         /*
  280.          * Sanitize naughty HTML elements
  281.          *
  282.          * If a tag containing any of the words in the list
  283.          * below is found, the tag gets converted to entities.
  284.          *
  285.          * So this: <blink>
  286.          * Becomes: &lt;blink&gt;
  287.          */
  288.         $naughty = 'alert|prompt|confirm|applet|audio|basefont|base|behavior|bgsound|blink|body|embed|expression|form|frameset|frame|head|html|ilayer|iframe|input|button|select|isindex|layer|link|meta|keygen|object|plaintext|style|script|textarea|title|math|video|svg|xml|xss';
  289.         $str = preg_replace_callback('#<(/*\s*)(' . $naughty . ')([^><]*)([><]*)#is', array($this, '_sanitize_naughty_html'), $str);
  290.  
  291.         /*
  292.          * Sanitize naughty scripting elements
  293.          *
  294.          * Similar to above, only instead of looking for
  295.          * tags it looks for PHP and JavaScript commands
  296.          * that are disallowed.  Rather than removing the
  297.          * code, it simply converts the parenthesis to entities
  298.          * rendering the code un-executable.
  299.          *
  300.          * For example:    eval('some code')
  301.          * Becomes:        eval&#40;'some code'&#41;
  302.          */
  303.         $str = preg_replace(
  304.                 '#(alert|prompt|confirm|cmd|passthru|eval|exec|expression|system|fopen|fsockopen|file|file_get_contents|readfile|unlink)(\s*)\((.*?)\)#si', '\\1\\2&#40;\\3&#41;', $str
  305.         );
  306.  
  307.         // Final clean up
  308.         // This adds a bit of extra precaution in case
  309.         // something got through the above filters
  310.         $str = $this->_do_never_allowed($str);
  311.  
  312.         /*
  313.          * Images are Handled in a Special Way
  314.          * - Essentially, we want to know that after all of the character
  315.          * conversion is done whether any unwanted, likely XSS, code was found.
  316.          * If not, we return TRUE, as the image is clean.
  317.          * However, if the string post-conversion does not matched the
  318.          * string post-removal of XSS, then it fails, as there was unwanted XSS
  319.          * code found and removed/changed during processing.
  320.          */
  321.  
  322.         if ($is_image === TRUE)
  323.         {
  324.             return ($str === $converted_string);
  325.         }
  326.  
  327.  
  328.         return $str;
  329.     }

  331.     public function remove_invisible_characters($str, $url_encoded = TRUE)
  332.     {
  333.         $non_displayables = array();
  334.  
  335.         // every control character except newline (dec 10)
  336.         // carriage return (dec 13), and horizontal tab (dec 09)
  337.  
  338.         if ($url_encoded)
  339.         {
  340.             $non_displayables[] = '/%0[0-8bcef]/'; // url encoded 00-08, 11, 12, 14, 15
  341.             $non_displayables[] = '/%1[0-9a-f]/'; // url encoded 16-31
  342.         }
  343.  
  344.         $non_displayables[] = '/[\x00-\x08\x0B\x0C\x0E-\x1F\x7F]+/S'; // 00-08, 11, 12, 14-31, 127
  345.  
  346.         do
  347.         {
  348.             $str = preg_replace($non_displayables, '', $str, -1, $count);
  349.         } while ($count);
  350.  
  351.         return $str;
  352.     }

  354.     protected function _remove_evil_attributes($str, $is_image)
  355.     {
  356.         // All javascript event handlers (e.g. onload, onclick, onmouseover), style, and xmlns
  357.         $evil_attributes = array('on\w*', 'style', 'xmlns', 'formaction', 'form', 'xlink:href', 'FSCommand', 'seekSegmentTime');
  358.  
  359.         if ($is_image === TRUE)
  360.         {
  361.             /*
  362.              * Adobe Photoshop puts XML metadata into JFIF images,
  363.              * including namespacing, so we have to allow this for images.
  364.              */
  365.             unset($evil_attributes[array_search('xmlns', $evil_attributes)]);
  366.         }
  367.  
  368.         do
  369.         {
  370.             $count = 0;
  371.             $attribs = array();
  372.  
  373.             // find occurrences of illegal attribute strings with quotes (042 and 047 are octal quotes)
  374.             preg_match_all('/(?<!\w)(' . implode('|', $evil_attributes) . ')\s*=\s*(\042|\047)([^\\2]*?)(\\2)/is', $str, $matches, PREG_SET_ORDER);
  375.  
  376.             foreach ($matches as $attr)
  377.             {
  378.                 $attribs[] = preg_quote($attr[0], '/');
  379.             }
  380.  
  381.             // find occurrences of illegal attribute strings without quotes
  382.             preg_match_all('/(?<!\w)(' . implode('|', $evil_attributes) . ')\s*=\s*([^\s>]*)/is', $str, $matches, PREG_SET_ORDER);
  383.  
  384.             foreach ($matches as $attr)
  385.             {
  386.                 $attribs[] = preg_quote($attr[0], '/');
  387.             }
  388.  
  389.             // replace illegal attribute strings that are inside an html tag
  390.             if (count($attribs) > 0)
  391.             {
  392.                 $str = preg_replace('/(<?)(\/?[^><]+?)([^A-Za-z<>\-])(.*?)(' . implode('|', $attribs) . ')(.*?)([\s><]?)([><]*)/i', '$1$2 $4$6$7$8', $str, -1, $count);
  393.             }
  394.         } while ($count);
  395.  
  396.         return $str;
  397.     }

  402.                 // encode captured opening or closing brace to prevent recursive vectors:
  403.                 . str_replace(array('>', '<'), array('&gt;', '&lt;'), $matches[4]);
  404.     }

  406.     protected function _js_link_removal($match)
  407.     {
  408.         return str_replace(
