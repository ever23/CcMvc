recibe los archivos enviados por el navegador

  1.  <?php
  2.  // cuando se recible un archivo mediante post
  3.  $file= new PostFiles('miFile');
  4.  if($file->is_Uploaded())
  5.  {
  6.       $file->Copy('C:/Xampp/htdocs/miapp/miFile.txt');
  7.  }
  8.  ?>
En caso de esperar varios archivos
  1.  <?php
  2.  // cuando se recible varios archivos mediante post
  3.  $Files= new PostFiles('miFile');
  4.  if($Files->is_Uploaded())
  5.  {
  6.       foreach($Files as $f)
  7.       {
  8.           $f->Copy('C:/Xampp/htdocs/miapp/'.$f['name']);
  9.       }
  10.  }
  11.  ?>
